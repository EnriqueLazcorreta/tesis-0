select $c = \{p.item_1,p.item_2,\ldots,p.item_{k-1},q.item_{k-1}\}$
from $L_{k-1}p,L_{k-1}q$
where  $(p.item_1 = q.item_1,\ldots,p.item_{k-2} = q.item_{k-2},$
          $p.item_{k-1} < q.item_{k-1})$
        and
          forall ($k-1$)-subsets $s$ of $c$
		        $s \in \aprioriL[k-1]$;
